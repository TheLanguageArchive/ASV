from 1.0 to >= 1.1
	- add new context parameter nl.mpi.translationservicebase

		e.g.  <Parameter name="nl.mpi.translationservicebase"
						 value="https://lux16.mpi.nl/ds/TranslationService" />       

	- add new context parameters nl.mpi.metadatabrowser.adapter.CSDBConnectionResource
 							     nl.mpi.metadatabrowser.adapter.AMS2ConnectionResource
 							     nl.mpi.metadatabrowser.adapter.DAMLRResource
 							     nl.mpi.metadatabrowser.adapter.ADSResource
				 
		e.g. <Parameter name="nl.mpi.metadatabrowser.adapter.CSDBConnectionResource"
					    value="java:comp/env/jdbc/CSDB" />   
					    
			<Parameter name="nl.mpi.metadatabrowser.adapter.AMS2ConnectionResource"
					    value="java:comp/env/jdbc/AMS2DB" />   

			<Parameter name="nl.mpi.metadatabrowser.adapter.DAMLRResource"
					    value="java:comp/env/ldap/DAMLR" />   
			
			<Parameter name="nl.mpi.metadatabrowser.adapter.ADSResource"
					    value="java:comp/env/kdc/MPI-NL" />  
					    
	- add new context parameters nl.mpi.metadatabrowser.schemeReplacePrefix and
								 nl.mpi.metadatabrowser.schemeReplaceTargetScheme
				(which replace nl.mpi.imdiBrowser.forceHttpOrHttps and
							   nl.mpi.imdiBrowser.forceHttpsPrefix)
						 
      e.g.:
      <Parameter 
         name="nl.mpi.metadatabrowser.schemeReplacePattern"
         value="http:\/\/lux16\.mpi\.nl.*"
         description="URLs matching this regex will have their scheme rewritten" />
         
      <Parameter 
         name="nl.mpi.metadatabrowser.schemeReplaceTargetScheme"
         value="https"
         description="protocol enforcement on matching URLs, e.g. http or https" />
         
	- (optional) add new optional context parameters 
				nl.mpi.metadatabrowser.wicket.maxSizePerSession and
				nl.mpi.metadatabrowser.wicket.inmemoryCacheSize
	
	  e.g. (with default values)
	  
	  <Parameter 
	    name="nl.mpi.metadatabrowser.wicket.maxSizePerSession"
		value="10M" 
		description="Representation of the size for page store file (cache on disk per
					 session), e.g. '10M' or '1024K'"/>   

	  <Parameter 
	    name="nl.mpi.metadatabrowser.wicket.inmemoryCacheSize"
		value="0" 
		description="The integer parameter is the maximum number of page instances that 
     				 will be saved into application-scoped cache (0 is disabled)"/>
